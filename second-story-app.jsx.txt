// Mock multiple photos per item using color variations
function getPhotos(item) {
  return [
    { bg: `linear-gradient(135deg,${item.color}33,${item.color}66)`, emoji: item.emoji },
    { bg: `linear-gradient(160deg,${item.color}22,${item.color}88)`, emoji: item.emoji },
    { bg: `linear-gradient(120deg,${item.color}44,${item.color}33)`, emoji: item.emoji },
    { bg: `linear-gradient(180deg,${item.color}55,${item.color}22)`, emoji: item.emoji },
  ];
}

function PhotoCarousel({ item, saved, onSave, onBack }) {
  const photos = getPhotos(item);
  const [current, setCurrent] = useState(0);
  const startX = useRef(null);
  const trackRef = useRef(null);

  const goTo = (idx) => setCurrent(Math.max(0, Math.min(photos.length - 1, idx)));

  const onTouchStart = (e) => { startX.current = e.touches[0].clientX; };
  const onTouchEnd = (e) => {
    if (startX.current === null) return;
    const diff = startX.current - e.changedTouches[0].clientX;
    if (Math.abs(diff) > 40) goTo(current + (diff > 0 ? 1 : -1));
    startX.current = null;
  };

  // Mouse drag support for desktop
  const mouseStart = useRef(null);
  const onMouseDown = (e) => { mouseStart.current = e.clientX; };
  const onMouseUp = (e) => {
    if (mouseStart.current === null) return;
    const diff = mouseStart.current - e.clientX;
    if (Math.abs(diff) > 40) goTo(current + (diff > 0 ? 1 : -1));
    mouseStart.current = null;
  };

  return (
    <div style={{position:"relative",height:300,overflow:"hidden",userSelect:"none"}}
      onTouchStart={onTouchStart} onTouchEnd={onTouchEnd}
      onMouseDown={onMouseDown} onMouseUp={onMouseUp}
    >
      {/* Slides track */}
      <div ref={trackRef} style={{display:"flex",height:"100%",transition:"transform 0.35s cubic-bezier(0.25,0.46,0.45,0.94)",transform:`translateX(-${current * 100}%)`}}>
        {photos.map((photo, i) => (
          <div key={i} style={{minWidth:"100%",height:"100%",background:photo.bg,display:"flex",alignItems:"center",justifyContent:"center",fontSize:110,flexShrink:0}}>
            {photo.emoji}
          </div>
        ))}
      </div>

      {/* Gradient overlay bottom */}
      <div style={{position:"absolute",bottom:0,left:0,right:0,height:80,background:"linear-gradient(to top, rgba(0,0,0,0.35), transparent)",pointerEvents:"none"}}/>

      {/* Back button */}
      <div style={{position:"absolute",top:52,left:16}}>
        <BackBtn onPress={onBack} light={false}/>
      </div>

      {/* Save button */}
      <button onClick={onSave} style={{position:"absolute",top:52,right:16,width:38,height:38,borderRadius:"50%",background:"rgba(255,255,255,0.92)",border:"none",fontSize:20,cursor:"pointer",display:"flex",alignItems:"center",justifyContent:"center",boxShadow:"0 2px 8px rgba(0,0,0,0.15)"}}>
        {saved ? "❤️" : "🤍"}
      </button>

      {/* Boosted badge */}
      {item.boosted && (
        <div style={{position:"absolute",top:52,left:64,background:C.orange,color:"white",fontSize:10,fontWeight:700,letterSpacing:"1px",textTransform:"uppercase",padding:"5px 12px",borderRadius:20}}>
          Boosted
        </div>
      )}

      {/* Photo counter top-right area */}
      <div style={{position:"absolute",bottom:16,right:16,background:"rgba(0,0,0,0.45)",borderRadius:20,padding:"4px 12px",display:"flex",alignItems:"center",gap:6}}>
        <svg width="12" height="12" viewBox="0 0 24 24" fill="none">
          <rect x="3" y="5" width="18" height="14" rx="3" stroke="white" strokeWidth="1.8"/>
          <circle cx="12" cy="12" r="3.5" stroke="white" strokeWidth="1.8"/>
          <circle cx="18.5" cy="7.5" r="1" fill="white"/>
        </svg>
        <span style={{color:"white",fontSize:11,fontWeight:700}}>{current+1} / {photos.length}</span>
      </div>

      {/* Dot indicators */}
      <div style={{position:"absolute",bottom:16,left:"50%",transform:"translateX(-50%)",display:"flex",gap:6,alignItems:"center"}}>
        {photos.map((_,i) => (
          <div key={i} onClick={()=>goTo(i)} style={{width:i===current?20:6,height:6,borderRadius:3,background:i===current?"white":"rgba(255,255,255,0.45)",transition:"all 0.3s",cursor:"pointer"}}/>
        ))}
      </div>

      {/* Left / Right tap zones (subtle) */}
      {current > 0 && (
        <button onClick={()=>goTo(current-1)} style={{position:"absolute",left:0,top:0,bottom:0,width:"20%",background:"transparent",border:"none",cursor:"pointer"}}/>
      )}
      {current < photos.length - 1 && (
        <button onClick={()=>goTo(current+1)} style={{position:"absolute",right:0,top:0,bottom:0,width:"20%",background:"transparent",border:"none",cursor:"pointer"}}/>
      )}
    </div>
  );
}

const CATS_LIST = ["Furniture","Electronics","Clothing","Sports","Books","Vehicles","Appliances","Other"];
const CONDITIONS = ["Like New","Good","Fair","Heavily Used"];

function emptyItem() {
  return { id: Date.now() + Math.random(), title:"", price:"", cat:"Electronics", condition:"Good", desc:"", photo:null, usedConsent:false, errors:{} };
}

function BulkListing() {
  const [isBulk, setIsBulk] = useState(false);
  const [items, setItems] = useState([emptyItem()]);
  const [published, setPublished] = useState(false);
  const [globalConsent, setGlobalConsent] = useState(false);

  const update = (id, field, val) => {
    setItems(prev => prev.map(i => i.id===id ? {...i, [field]:val, errors:{...i.errors,[field]:""}} : i));
  };

  const addItem = () => setItems(prev => [...prev, emptyItem()]);

  const removeItem = (id) => {
    if(items.length===1) return;
    setItems(prev => prev.filter(i => i.id!==id));
  };

  const validate = () => {
    let valid = true;
    setItems(prev => prev.map(item => {
      const errors = {};
      if(!item.title.trim()) { errors.title = "Title required"; valid=false; }
      if(!item.price || isNaN(item.price) || +item.price<=0) { errors.price = "Valid price required"; valid=false; }
      if(!item.usedConsent) { errors.usedConsent = "Must confirm item is used"; valid=false; }
      return {...item, errors};
    }));
    if(!globalConsent) valid = false;
    return valid;
  };

  const publish = () => {
    if(validate()) setPublished(true);
  };

  // SUCCESS SCREEN
  if(published) return (
    <div style={{padding:"32px 0",textAlign:"center"}}>
      <div style={{fontSize:64,marginBottom:16}}>🎉</div>
      <div style={{fontSize:20,fontWeight:900,color:C.ink,marginBottom:8}}>
        {items.length===1 ? "Item Listed!" : `${items.length} Items Listed!`}
      </div>
      <div style={{fontSize:14,color:C.inkLight,marginBottom:24,lineHeight:1.7}}>
        Your {items.length===1?"listing is":"listings are"} now live and visible to buyers in your area.
      </div>
      <div style={{display:"flex",flexDirection:"column",gap:10,marginBottom:24}}>
        {items.map((item,i)=>(
          <div key={item.id} style={{background:C.white,borderRadius:14,padding:"12px 16px",display:"flex",alignItems:"center",gap:12,border:`1px solid ${C.border}`,textAlign:"left"}}>
            <div style={{width:40,height:40,borderRadius:10,background:C.greenLight,display:"flex",alignItems:"center",justifyContent:"center",fontSize:20,flexShrink:0}}>📦</div>
            <div style={{flex:1}}>
              <div style={{fontWeight:700,fontSize:13,color:C.ink}}>{item.title||`Item ${i+1}`}</div>
              <div style={{fontSize:12,color:C.inkLight}}>₦{Number(item.price).toLocaleString()} · {item.cat}</div>
            </div>
            <span style={{fontSize:10,fontWeight:700,padding:"3px 10px",borderRadius:20,background:`${C.green}15`,color:C.green,textTransform:"uppercase"}}>Live</span>
          </div>
        ))}
      </div>
      <button onClick={()=>{setPublished(false);setItems([emptyItem()]);setIsBulk(false);setGlobalConsent(false);}}
        style={{width:"100%",background:C.orange,border:"none",borderRadius:14,padding:14,color:"white",fontWeight:800,fontSize:15,cursor:"pointer"}}>
        + List More Items
      </button>
    </div>
  );

  return (
    <div style={{paddingBottom:16}}>

      {/* ── USED ITEMS POLICY BANNER ── */}
      <div style={{background:"#0A0A0A",borderRadius:16,padding:16,marginBottom:16,display:"flex",gap:12,alignItems:"flex-start"}}>
        <div style={{width:36,height:36,borderRadius:10,background:`${C.orange}22`,border:`1px solid ${C.orange}44`,display:"flex",alignItems:"center",justifyContent:"center",fontSize:20,flexShrink:0}}>♻️</div>
        <div>
          <div style={{color:"white",fontWeight:800,fontSize:14,marginBottom:4}}>Second Story is for used items only</div>
          <div style={{color:"rgba(255,255,255,0.55)",fontSize:12,lineHeight:1.65}}>
            Every item listed here must be a pre-owned, second-hand item. New or counterfeit goods are not permitted and may result in account suspension.
          </div>
        </div>
      </div>

      {/* ── SINGLE / BULK TOGGLE ── */}
      <div style={{background:C.white,borderRadius:14,padding:4,marginBottom:16,display:"flex",border:`1px solid ${C.border}`}}>
        <button onClick={()=>{setIsBulk(false);setItems(prev=>[prev[0]||emptyItem()]);}} style={{flex:1,padding:"10px 0",border:"none",borderRadius:10,cursor:"pointer",background:!isBulk?C.green:"transparent",color:!isBulk?"white":C.inkLight,fontWeight:!isBulk?800:400,fontSize:14,transition:"all 0.2s",display:"flex",alignItems:"center",justifyContent:"center",gap:8}}>

      </div>

  );
}
